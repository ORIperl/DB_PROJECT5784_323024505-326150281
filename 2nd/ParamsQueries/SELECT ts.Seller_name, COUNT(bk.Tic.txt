SELECT 
    ts.Seller_name,
    ts.Seller_id,
    bk.Booking_id,
    t.Ticket_id,
    t.Ticket_type,
    t.Ticket_price,
    bk.Booking_date
FROM 
    TicketSeller ts
JOIN 
    Booking bk ON ts.Seller_id = bk.Seller_id
JOIN 
    Ticket t ON bk.Ticket_id = t.Ticket_id
WHERE 
    ts.Seller_name = &name
ORDER BY 
    bk.Booking_date DESC;
